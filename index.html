// Voice Chat Frontend - Real-time Continuous Mode // Implements VAD (Voice Activity Detection) for seamless voice conversations const BACKEND_URL = 'https://voice-chat-backend-production.up.railway.app'; const WS_URL = BACKEND_URL.replace('http', 'ws') + '/ws/voice'; // Recording state let ws = null; let wsConnected = false; let isRecording = false; let mediaRecorder = null; let audioContext = null; let analyser = null; let audioChunks = []; let silenceTimer = null; let isProcessing = false; // Track if we're waiting for server response // VAD settings const SILENCE_THRESHOLD = 5; // Audio level below this is silence (0-255) const SILENCE_DURATION = 2000; // 2 seconds of silence triggers processing let lastAudioTime = 0; // DOM elements const chatContainer = document.getElementById('chatContainer'); function addMessage(type, text) { const msg = document.createElement('div'); msg.className = `message ${type}`; msg.textContent = text; chatContainer.appendChild(msg); chatContainer.scrollTop = chatContainer.scrollHeight; } function clearChat() { chatContainer.innerHTML = ''; } function startCall() { clearChat(); addMessage('system', '🔄 連接中...'); ws = new WebSocket(WS_URL); ws.onopen = () => { console.log('[DEBUG WS] ✅ WebSocket connected'); addMessage('system', '✅ 已連接，開始監聽...'); wsConnected = true; isRecording = true; isProcessing = false; startRecording(); }; ws.onmessage = (event) => { try { const msg = JSON.parse(event.data); console.log('[DEBUG WS] 📥 Received type:', msg.type); switch (msg.type) { case 'transcription': console.log('[DEBUG 1] 🎤 Transcription:', msg.text); addMessage('system', `🎤 辨識: ${msg.text}`); addMessage('user', `👤 ${msg.text}`); break; case 'response': console.log('[DEBUG 2] 🤖 LLM Response:', msg.response); addMessage('system', `🤖 回覆: ${msg.response}`); addMessage('ai', `🤖 ${msg.response}`); break; case 'audio': console.log('[DEBUG 3] 🔊 TTS audio received, size:', msg.data ? msg.data.length : 0); if (msg.data && msg.data.length > 1000) { addMessage('system', `🔊 播放音頻 (${msg.data.length} bytes)`); const audio = document.getElementById('responseAudio'); audio.src = 'data:audio/mp3;base64,' + msg.data; audio.style.display = 'block'; audio.play().then(() => { console.log('[DEBUG 3] 🔊 Audio playing, restarting VAD monitoring...'); // Audio finished playing, restart recording cycle setTimeout(() => { isProcessing = false; if (wsConnected && isRecording) { console.log('[DEBUG VAD] 🔄 Restarting recording after audio playback'); restartRecording(); } }, 500); }).catch(err => { console.error('[DEBUG 3] ❌ Audio play error:', err); addMessage('system', `❌ 播放失敗`); // Still restart recording on error isProcessing = false; if (wsConnected && isRecording) restartRecording(); }); } else { console.error('[DEBUG 3] ❌ TTS audio too small:', msg.data ? msg.data.length : 0); addMessage('system', `❌ TTS 無效`); // Still restart recording on error isProcessing = false; if (wsConnected && isRecording) restartRecording(); } break; case 'reset': console.log('[DEBUG WS] 🔄 Conversation reset'); break; case 'error': console.error('[DEBUG WS] ❌ Error from server:', msg.message); addMessage('system', `❌ 錯誤: ${msg.message}`); // Restart recording after error setTimeout(() => { isProcessing = false; if (wsConnected && isRecording) restartRecording(); }, 1000); break; } } catch (e) { console.error('[DEBUG WS] ❌ Failed to parse message:', e); } }; ws.onerror = (error) => { console.error('[DEBUG WS] ❌ WebSocket error:', error); addMessage('system', '❌ 連接錯誤'); stopCall(); }; ws.onclose = () => { console.log('[DEBUG WS] 🔌 WebSocket closed'); addMessage('system', '🔌 連接已結束'); wsConnected = false; isRecording = false; stopVADMonitoring(); }; } function stopCall() { console.log('[DEBUG] 🛑 stopCall() called'); isRecording = false; stopVADMonitoring(); if (mediaRecorder && mediaRecorder.state === 'recording') { mediaRecorder.stop(); } if (ws && wsConnected) { ws.send(JSON.stringify({ type: 'reset' })); ws.close(); wsConnected = false; } } async function startRecording() { try { console.log('[DEBUG RECORD] 🎙️ Starting recording...'); const stream = await navigator.mediaDevices.getUserMedia({ audio: true }); // Set up audio context for VAD audioContext = new AudioContext(); const source = audioContext.createMediaStreamSource(stream); analyser = audioContext.createAnalyser(); analyser.fftSize = 256; analyser.smoothingTimeConstant = 0.3; source.connect(analyser); // Try different audio formats for Whisper compatibility const mimeTypes = ['audio/mpeg', 'audio/ogg', 'audio/webm;codecs=opus', 'audio/webm']; let mimeType = mimeTypes.find(t => MediaRecorder.isTypeSupported(t)) || 'audio/webm'; console.log('[DEBUG RECORD] 📋 Using MIME type:', mimeType); mediaRecorder = new MediaRecorder(stream, { mimeType }); audioChunks = []; mediaRecorder.ondataavailable = event => { console.log('[DEBUG RECORD] 📦 Audio chunk received, size:', event.data.size); if (event.data.size > 0) { audioChunks.push(event.data); } }; mediaRecorder.onstop = () => { console.log('[DEBUG RECORD] 🛑 Recording stopped, chunks:', audioChunks.length); stream.getTracks().forEach(track => track.stop()); }; mediaRecorder.start(1000); // Collect data every second console.log('[DEBUG RECORD] ✅ Recording started'); addMessage('system', '🎙️ 開始說話...'); // Start VAD monitoring startVADMonitoring(stream); } catch (error) { console.error('[DEBUG RECORD] ❌ Failed to start recording:', error); addMessage('system', '❌ 無法訪問麥克風: ' + error.message); stopCall(); } } function startVADMonitoring(stream) { console.log('[DEBUG VAD] 🔍 Starting VAD monitoring...'); stopVADMonitoring(); // Create a separate audio context for VAD if not already created if (!audioContext) { audioContext = new AudioContext(); const source = audioContext.createMediaStreamSource(stream); analyser = audioContext.createAnalyser(); analyser.fftSize = 256; analyser.smoothingTimeConstant = 0.3; source.connect(analyser); } const dataArray = new Uint8Array(analyser.frequencyBinCount); let silenceStart = null; silenceTimer = setInterval(() => { if (!isRecording || isProcessing) return; analyser.getByteFrequencyData(dataArray); // Calculate average audio level const sum = dataArray.reduce((a, b) => a + b, 0); const average = sum / dataArray.length; console.log('[DEBUG VAD] 📊 Audio level:', average.toFixed(1), average < SILENCE_THRESHOLD ? '(silence)' : '(speech)'); if (average < SILENCE_THRESHOLD) { // Silence detected if (!silenceStart) { silenceStart = Date.now(); console.log('[DEBUG VAD] 🔇 Silence started at:', new Date().toLocaleTimeString()); } else { const silenceDuration = Date.now() - silenceStart; if (silenceDuration >= SILENCE_DURATION) { console.log('[DEBUG VAD] ✅ Silence for 2s detected, triggering send!'); silenceStart = null; silenceDetected(); } } } else { // Audio detected if (silenceStart) { console.log('[DEBUG VAD] 🔊 Speech detected, resetting silence timer'); } silenceStart = null; lastAudioTime = Date.now(); } }, 200); // Check every 200ms } function stopVADMonitoring() { if (silenceTimer) { clearInterval(silenceTimer); silenceTimer = null; console.log('[DEBUG VAD] 🛑 VAD monitoring stopped'); } } function silenceDetected() { if (isProcessing) { console.log('[DEBUG VAD] ⏳ Already processing, skipping'); return; } console.log('[DEBUG VAD] 📤 Silence detected, preparing to send audio'); if (audioChunks.length === 0) { console.log('[DEBUG VAD] ⚠️ No audio chunks accumulated'); return; } isProcessing = true; // Stop current recording if (mediaRecorder && mediaRecorder.state === 'recording') { mediaRecorder.stop(); } // Get the MIME type from mediaRecorder const mimeType = mediaRecorder ? mediaRecorder.mimeType : 'audio/webm'; // Combine all chunks const audioBlob = new Blob(audioChunks, { type: mimeType }); const reader = new FileReader(); reader.onload = () => { const base64 = reader.result.split(',')[1]; console.log('[DEBUG VAD] 📤 Sending audio, chunks:', audioChunks.length, 'base64 length:', base64.length); addMessage('system', `📤 發送音頻...`); if (ws && wsConnected) { ws.send(JSON.stringify({ type: 'audio', data: base64, language: 'zh' })); } else { console.error('[DEBUG VAD] ❌ WS not connected'); isProcessing = false; } }; reader.readAsDataURL(audioBlob); // Reset for next recording cycle audioChunks = []; } function restartRecording() { console.log('[DEBUG RECORD] 🔄 Restarting recording...'); // Small delay before restarting setTimeout(() => { if (isRecording && wsConnected && !isProcessing) { startRecording(); } }, 300); } // Event listeners document.getElementById('callBtn').addEventListener('click', () => { if (isRecording) { console.log('[DEBUG] 🛑 Call button clicked - stopping'); stopCall(); } else { console.log('[DEBUG] 📞 Call button clicked - starting'); startCall(); } }); document.getElementById('hangupBtn').addEventListener('click', stopCall); // Audio element ended event - restart recording after audio finishes document.getElementById('responseAudio').addEventListener('ended', () => { console.log('[DEBUG] 🔊 Audio playback ended'); }); // Initialize console.log('[DEBUG] 🚀 Voice Chat Frontend initialized'); addMessage('system', '🎙️ 點擊麥克風開始持續語音對話');
